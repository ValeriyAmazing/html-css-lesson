    transition: all 0.3s ease;
}

    transform: scale(1.1);
    box-shadow: 0 20px 40px rgba(0,0,0,0.3);
}

/* Вращающийся спиннер */
@keyframes spin {
    from { transform: rotate(0deg); }
    to { transform: rotate(360deg); }
}
.spinner {
    animation: spin 1s linear infinite;
}

/* Пульсация */
@keyframes pulse {
    0%, 100% { transform: scale(1); }
    50% { transform: scale(1.05); }
}
